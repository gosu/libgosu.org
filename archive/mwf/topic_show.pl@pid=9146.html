I solved my second problem with the multiplication of complex numbers.
Perhaps there is a simpler solution...

def draw
      @img.draw_rot(@shape.body.p.x, @shape.body.p.y, 2, @shape.body.a.radians_to_gosu)
      @rotate_shape_array = @shape_array.map { |vect|
         c1 = Complex(vect.x, vect.y)
         c2 = Complex.polar(1, @shape.body.a)
         c3 = c1*c2
         vect_result = CP::Vec2.new(c3.real, c3.imag)}
      def drawline j, k
      Gosu.draw_line(@shape.body.p.x + @rotate_shape_array[j].x,
                     @shape.body.p.y + @rotate_shape_array[j].y,
