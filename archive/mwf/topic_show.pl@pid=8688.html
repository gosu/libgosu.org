Here's a small wrapper

class Map
  class Layer
    def initialize(data,file)
      @data = data
      @xfactor = 1
      @yfactor = 1
      @xoffset = 0
      @yoffset = 0
      @image = Gosu::Image.new(file)
    end
    def draw(x,y)
      @image.draw(x*@xfactor+@xoffset,y*@yfactor+@yoffset,1)
    end
  end
  def initialize(name)
    @data = Gosu::Tiled.load_json(self, name+'.json')
    @layers = []
    @data.layers.select{ |_| _.type=='tilelayer'}.each_with_index do |data,index|
      @layers.push(Layer.new(data,name+'_'+(index+1).to_s+'.map'))
    end
  end
  def draw(x,y)
    @layers.each do |layer|
      layer.draw(x,y)
    end
  end
end

Here's a batch file to export Tiled maps

@echo off
setlocal ENABLEDELAYEDEXPANSION
for /f "tokens=3 delims== "  %%n in ('findstr /c:"layer name" %1.tmx') do (
set layers=!layers! %%~n
)
set c=1
for %%l in (%layers::= %) do (
    set hide=
    for %%h in (%layers::= %) do (
      if not %%h==%%l (
         set hide= !hide! --hide-layer %%h
      )
    )
    call "C:/program files (x86)/tiled/tmxrasterizer.exe" !hide! %1.tmx %%l_.map
    copy %%l_.map %1_!c!.map
    set /a c+=1
)
